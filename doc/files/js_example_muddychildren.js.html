/**
 * @param truePropositions an array of true propositions
 * @returns a state where the public channel is empty and truePropositions give the valuation
 * @example new AsynchronousPublicChannelState(["p", "q"])
 * */
function MuddyChildrenWorld(truePropositions)
{
    this.valuation = new Valuation(truePropositions);

    this.modelCheck = function(p)
    {
          return this.valuation.modelCheck(p);
    }


    this.toString = function()
    {
      return this.valuation.toString();
    }



    this.draw = function(context)
    {
      context.beginPath();
      context.moveTo(0,0);
      context.lineTo(80+64,0);
      context.lineTo(80+64+16, 64);
      context.lineTo(-16, 64);

      context.fillStyle = worldBackgroundColor;
      context.fill();



        context.drawImage(agent1img, 0, 0, 64, 64);
        context.drawImage(agent2img, agent2x, 0, 64, 64);

        context.font="12px Verdana";
        context.strokeText("a", 16, 48);
        context.strokeText("b", agent2x+32, 48);

        if(this.modelCheck("ma"))
        {
          context.drawImage(mud, 16, 0, 32, 32);
        }

        if(this.modelCheck("mb"))
        {
          context.drawImage(mud, agent2x+16, 0, 32, 32);
        }


    }

    this.getAgentRectangle = function(agentName)
    {
          if(agentName == "a")
               return new Rectangle(0, 0, 64, 64);
          else
               return new Rectangle(agent2x, 0, 64, 64);

    }

}















function setExampleMuddychildren()
{
  M = new EpistemicModel();
  w = "w";

  M.addWorld("w", new MuddyChildrenWorld(["ma", "mb"]));
  M.addWorld("u", new MuddyChildrenWorld(["mb"]));
  M.addWorld("s", new MuddyChildrenWorld(["ma"]));
  M.addWorld("t", new MuddyChildrenWorld([""]));

  M.addEdge("a", "w", "u");
  M.addEdge("a", "u", "w");

  M.addEdge("a", "s", "t");
  M.addEdge("a", "t", "s");

  M.addEdge("b", "w", "s");
  M.addEdge("b", "s", "w");

  M.addEdge("b", "u", "t");
  M.addEdge("b", "t", "u");

  M.makeReflexiveRelation("a");
  M.makeReflexiveRelation("b");
