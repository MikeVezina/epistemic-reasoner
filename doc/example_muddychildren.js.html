'use strict';

/**
 * @param truePropositions an array of true propositions. Proposition "ma" stands for "a is muddy" and "mb" stands for "b is muddy".
 * @returns a state corresponding to the muddy children puzzle
 * @example new MuddyChildrenWorld(["ma", "mb"])
 * */
class MuddyChildrenWorld extends WorldValuation
{
    constructor(propositions)
    {
        super(propositions);
        this.agentPos["a"] = {x: 32, y: 32, r: 32};
        this.agentPos["b"] = {x: 68+32, y: 32, r: 32};
        this.agentPos["c"] = undefined;
    }

    draw(context)
    {
        this.drawAgents(context);

        if(this.modelCheck("ma"))
         context.drawImage(mud, 16, 0, 32, 16);

       if(this.modelCheck("mb"))
         context.drawImage(mud, this.agentPos["b"].x-16, 0, 32, 16);
    }

}

MuddyChildrenWorld.mud = new Image();
MuddyChildrenWorld.mud.src = "img/mud.png";




function getMuddyChildrenInitialEpistemicModel()
{
  let M = new EpistemicModel();

  M.addWorld("w", new MuddyChildrenWorld(["ma", "mb"]));
  M.addWorld("u", new MuddyChildrenWorld(["mb"]));
  M.addWorld("t", new MuddyChildrenWorld([""]));
  M.addWorld("s", new MuddyChildrenWorld(["ma"]));


  M.setPointedWorld("w");

  M.addEdgesCluster("a", ["w", "u"]);
  M.addEdgesCluster("a", ["s", "t"]);
  M.addEdgesCluster("b", ["w", "s"]);
  M.addEdgesCluster("b", ["t", "u"]);

  return M;
}




function setExampleMuddychildren()
{
 M = getMuddyChildrenInitialEpistemicModel();

 $('#panelExampleFormulas').html('Example of formulas: '
+ "<ul><li>(ma or mb)</li>" +
  "<li>((not (K a ma)) and (not (K b mb)))</li>" +
  "</ul>");

  addExplanation("Two children are playing in the grass. They see each other but they do not know whether they are muddy or not.",
                 "(not (K a (K b (ma or mb))))");

  addButtonAction(function() {performAction(actionMuddyChildrenFather());
            speak("c", "at least one of you is muddy");},
                 "Father says at least one of you is muddy",
               "(not (K a (K b (ma or mb))))");

 addExplanation("Now, both children learned that they are muddy. Could you figure it out?",
                "((K a ma) and (K b mb))");

  addButtonAction(function() {
          performAction(actionMuddyChildrenDoNotKnow());
          speak("ab", "I do not know whether I am muddy or not");
        },        "They say they do not know",
      "(not ((K a ma) and (K b mb)))");

  addButtonAction(function() {performAction(actionMuddyChildrenKnow());
              speak("ab", "I know that I am muddy");
          },         "They say they know",
        "((K a ma) and (K b mb))");

 compute();
 computeButtonsVisibleOrNot();

}

let actionMuddyChildrenFather = () => getActionModelPublicAnnouncement("(ma or mb)");
let actionMuddyChildrenDoNotKnow = () => getActionModelPublicAnnouncement("((not (K a ma)) and (not (K b mb)))");
let actionMuddyChildrenKnow = () => getActionModelPublicAnnouncement("((K a ma) and (K b mb))");
