EpistemicModel.prototype = new Graph();




/**
 * Create a new empty epistemic model
 * @class EpistemicModel
 * @constructor
   @extends Graph
 */
function EpistemicModel()
{
	this.nodes = new Array();
	this.successors = new Array();
	this.dotstyle = "[shape=circle, fillcolor=lightblue2, style=filled]";


	/**
	 * @method addWorld
	 * @param w world identifier
	 * @param content the content a world. It is an object that should
	     implement a method modelCheck for check atomic formulas.
			 Typically, it is an object containing a propositional World.
	 * */
	this.addWorld = function(w, content)
	{
		this.addNode(w, content);
	}



	/**
	 * @method modelCheck
	 * @param w world identifier
	 * @param phi a formula (internal representation of a formula)
	 * @returns true if formula phi is true in w
	 * */
	this.modelCheck = function(w, phi)
	{
		//console.log("modelCheck(", w, ", ", phi, ")");
		if(phi instanceof Array)
		{
		    if(phi[0] == "not")
		      return !this.modelCheck(w, phi[1]);
				else
		    if(phi[1] == "or")
				{
							for(var i = 0; i < phi.length; i+=2)
							{
											if(this.modelCheck(w, phi[i]))
													return true;
							}
							return false;
				}
				else
		    if(phi[1] == "and")
				{
					for(var i = 0; i < phi.length; i+=2)
					{
									if(!this.modelCheck(w, phi[i]))
											return false;
					}
					return true;
				}
				else
		    if((phi[0] == "K") || (phi[0] == "know"))
		    {
							let agent = phi[1];
							let psi = phi[2];

						/*	if(this.successors[agent] == undefined)
								console.log(agent);*/
							let successors = this.getSuccessors(w, agent);
						/*	console.log(successors);
							console.log(w);
							console.log(agent);*/

							for(let iu in successors)
							{
							    if(!this.modelCheck(successors[iu], psi))
												return false;

							}

							return true;



		    }
				else if((phi[0] == "Kpos") || (phi[0] == "knowpos"))
		    {
							let agent = phi[1];
							let psi = phi[2];

							let successors = this.getSuccessors(w, agent);
							for(let iu in successors)
							{
							    if(this.modelCheck(successors[iu], psi))
										return true;

							}

							return false;



		    }
		    else
		    {
		      return this.nodes[w].modelCheck(phi);
		    }



		}
		else if(phi == "top")
		  return true;
		else if(phi == "bottom")
			return false;
		else
		{
		    return this.nodes[w].modelCheck(phi);

		}


	}

}
