/**
 * @param truePropositions an array of true propositions
 * @returns a state where the public channel is empty and truePropositions give the valuation
 * @example new AsynchronousPublicChannelState(["p", "q"])
 * */
function SallyAndAnneWorld(truePropositions)
{
    this.valuation = new Valuation(truePropositions);

    this.agentAnnex = 96;
    this.agentSallyx = 32;


    this.modelCheck = function(p)
    {
          return this.valuation.modelCheck(p);
    }


    this.toString = function()
    {
      return this.valuation.toString();
    }



    this.draw = function(context)
    {
      context.beginPath();
      context.moveTo(0,0);
      context.lineTo(80+64,0);
      context.lineTo(80+64+16, 64);
      context.lineTo(-16, 64);

      context.fillStyle = worldBackgroundColor;
      context.fill();

      context.beginPath();
      context.fillStyle = "#FF0000";
      context.strokeStyle = "black";
      context.moveTo(32+(128-32)/2,0);
      context.lineTo(128,agentY);
      context.lineTo(32, agentY);
      context.fill();
      context.stroke();


      context.fillStyle = "#DDBB22";
      context.fillRect(32, agentY, 96, 64-agentY);
      context.strokeStyle = "black";
      context.strokeRect(32, agentY, 96, 64-agentY);

      context.strokeStyle = "#DDBB22";
      context.moveTo(32,agentY+16);
      context.lineTo(32,agentY+32);
      context.stroke();




        context.drawImage(agentAnneimg, this.agentAnnex, agentY, 32, 32);





        if(this.modelCheck("bspy"))
        {
              context.beginPath();
              context.setLineDash([5, 3]);
              context.strokeStyle = "black";
              context.moveTo(3, 16);
              context.lineTo(48, 36);
              context.stroke();
              context.setLineDash([]);

        }


        if(this.modelCheck("bhere"))
        {
          this.agentSallyx = 32;
          context.drawImage(agentSallyimg, this.agentSallyx, agentY, 32, 32);
        }

        else {
            this.agentSallyx = 0;
            context.drawImage(agentSallyimg, this.agentSallyx, agentY, 16, 16);
        }



        context.font="12px Verdana";
        context.strokeStyle = "black";
        context.strokeText("a", this.agentAnnex+8, agentY+16);
        context.strokeText("s", this.agentSallyx+16, agentY+12);


        context.drawImage(boxImg, 80, objectsY, 32, 32);
        context.drawImage(basketImg, 46, objectsY, 32, 32);

        if(this.modelCheck("marbleBasket"))
        {
          context.drawImage(basketWithMarbleImg, 46, objectsY, 32, 32);
        }

        if(this.modelCheck("marbleBox"))
        {
          context.drawImage(marbleImg, 90, objectsY+12, 8, 8);
        }


        if(this.modelCheck("marbleb"))
        {
          var marbleWidth  = 16;
          if(this.agentSallyx < 10)
            marbleWidth = 8;
          context.drawImage(marbleImg, this.agentSallyx, agentY+4, marbleWidth, marbleWidth);
        }

    }

    this.getAgentRectangle = function(agentName)
    {
          if(agentName == "a")
               return new Rectangle(this.agentAnnex, agentY, 32, 32);
          else
               return new Rectangle(this.agentSallyx, agentY, 32, 32);

    }

}

function actionSallyOutIn()
{
  var E = new ActionModel();


  if(M.getNode(w).modelCheck("bhere"))
          E.addAction("e", "top", new PropositionalAssignmentsPostcondition({"bhere": "bottom"}));
  else {
    E.addAction("e", "top", new PropositionalAssignmentsPostcondition({"bhere": "top", "bspy": "bottom"}));
  }
  E.makeReflexiveRelation("a");
  E.makeReflexiveRelation("b");

  return E;


}





function actionSallySpy()
{
  var E = new ActionModel();
    if(M.getNode(w).modelCheck("bhere"))
          throw "Sally should be out to spy / not spy the house";
    else if(M.getNode(w).modelCheck("bspy")) {
      E.addAction("e", "top", new PropositionalAssignmentsPostcondition({"bspy": "bottom"}));
    }
    else {
      E.addAction("e", "top", new PropositionalAssignmentsPostcondition({"bspy": "top"}));
    }
    E.addAction("f", "top", new TrivialPostCondition());

    E.addEdge("b", "e", "e");

    E.addEdge("a", "e", "f");
    E.addEdge("a", "f", "f");
    E.addEdge("b", "f", "f");

    return E;
}





function actionSallyMarbleSallyToMarbleBasket()
{
  var E =   new ActionModel();


  if(M.getNode(w).modelCheck("bhere"))
  if(M.getNode(w).modelCheck("marbleb"))
          E.addAction("e", "top", new PropositionalAssignmentsPostcondition({"marbleBasket": "top", "marbleb": "bottom"}));
  else {
    throw "Sally is outside";
  }


  E.makeReflexiveRelation("a");
  E.makeReflexiveRelation("b");

  return E;


}








function actionAnneTransfersMarbleFromBasketToBox()
{
  var E =   new ActionModel();

  if(!M.getNode(w).modelCheck("marbleBasket"))
  {
    throw "The marble should be in the basket";
  }

  if(M.getNode(w).modelCheck("bhere"))
  {
          E.addAction("e", "top", new PropositionalAssignmentsPostcondition({"marbleBasket": "bottom", "marbleBox": "top"}));

            E.makeReflexiveRelation("a");
            E.makeReflexiveRelation("b");
  }
  else
//Sally is outside
  if(M.getNode(w).modelCheck("bspy"))
  {
    var assignmentTransfer = new PropositionalAssignmentsPostcondition({"marbleBasket": "bottom", "marbleBox": "top"});
    E.addAction("e", "top", assignmentTransfer);
    E.addAction("f", "top", assignmentTransfer);
    E.addAction("t", "top", new TrivialPostCondition());

    E.addEdge("b", "e", "e");
    E.addEdge("a", "e", "f");

    E.addEdge("a", "f", "f");

    E.addEdge("b", "f", "t");
    E.addEdge("a", "t", "t");
    E.addEdge("b", "t", "t");
  }
  else
  {
          E.addAction("e", "top", new PropositionalAssignmentsPostcondition({"marbleBasket": "bottom", "marbleBox": "top"}));
          E.addAction("t", "top", new TrivialPostCondition());
          E.addEdge("a", "e", "e");

          E.addEdge("b", "e", "t");
          E.addEdge("a", "t", "t");
          E.addEdge("b", "t", "t");
  }



  return E;


}
