/**
 *@class Valuation
 * @constructor
 * @param truePropositions: an array of true propositions
 * @returns a valuation where all propositions in truePropositions are true and others are false
 */
function Valuation(truePropositions)
{
    this.propositions = new Array();

    if(truePropositions != undefined)
    {
      for(var i in truePropositions)
      {
	         this.propositions[truePropositions[i]] = true;
      }
    }


    
    this.modelCheck = function(p)
    {
    	if(this.propositions[p] == true)
    	  return true;
    	else
    	   return false;
    }


    this.toString = function()
    {
      var s = "";

      var putcomma = false;

	for(var i in this.propositions)
	{
	   if(this.propositions[i] == true)
	   {
	      if(putcomma)
		    s += ",";


		if(!putcomma)
		  putcomma = true;


		s += i;
	   }
	}


	return s;
    }



    this.toHTML = this.toString;
}








/**
 * @param a valuation
 * @returns a world that contains the valuation
 */
function PropositionalWorld(valuation)
{
  this.valuation = valuation;

  this.modelCheck = function(p)
  {
    return this.valuation.modelCheck(p);
  }

}
